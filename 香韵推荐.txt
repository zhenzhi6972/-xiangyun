let answers = {};

document.querySelectorAll('.options').forEach(group => {
  group.addEventListener('click', e => {
    if (!e.target.classList.contains('option')) return;
    const qid = group.id;
    group.querySelectorAll('.option').forEach(o => o.classList.remove('active'));
    e.target.classList.add('active');
    answers[qid] = e.target.dataset.value;
    checkReady();
  });
});

function checkReady() {
  document.getElementById('recommend').disabled = !(answers.q1 && answers.q2 && answers.q3);
}

document.getElementById('recommend').addEventListener('click', () => {
  const need = answers.q1;
  const level = answers.q2;
  const budget = answers.q3;

  let best = null, bestScore = -1;

  for (const [name, p] of Object.entries(products)) {
    let score = p.match[need] || 0;

    // 新手优先推入门款，老手可以推高端
    if (level === 'new' && ['清·自在','甜·好眠'].includes(name)) score += 2;
    if (level === 'expert' && ['醇·独处','韵·万象'].includes(name)) score += 2;

    // 预算匹配
    const priceNum = parseInt(p.price.replace(/[^0-9]/g,''));
    if (budget === 'low' && priceNum <= 200) score += 2;
    if (budget === 'mid' && priceNum > 200 && priceNum <= 400) score += 2;
    if (budget === 'high' && priceNum > 400) score += 2;

    // 降预算时的处理：选相对最接近的
    if (budget === 'low' && priceNum > 200) score -= 1;
    if (budget === 'mid' && priceNum > 400) score -= 1;

    if (score > bestScore) { bestScore = score; best = {name, ...p}; }
  }

  document.getElementById('r-name').textContent = best.name;
  document.getElementById('r-price').textContent = best.price;
  document.getElementById('r-desc').textContent = best.desc;
  document.getElementById('r-tags').innerHTML = best.tags.map(t => `<span class="tag">${t}</span>`).join('');
  document.getElementById('result').classList.add('show');
});
